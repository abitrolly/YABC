гадоў

.
Да

мінуўшчыны
,
старых

,

,

,

.
Вырашаюцца
многія

.
Бяроза
—

,

ў

Цікавы

і

.
Тут

,

на

увагу
.
Зазірніце
ў
Бярозаўскі
гісторыка-рэвалюцыйны

,
які
месціцца

вуліцы
Леніна
,
67
,
—

.

і

,

сярэдзіны

.
У

“
Літаратурныя

”
(

.
Кулажанка
,
А
.
Мальдзіс
,
С
.
Сачанка
)

:
“
Бярозаўскі

.

39
рукапісных
кніг
,
у
тым
ліку
11

,

2314

тамоў
.
Ад

,

,

сцяны

”
Узгадаў

і

,

ў

.
У

Міжнародная

:

і

.
Запланаваная

кастрычнік
,
гэтая

,

розных

асаблівую
ўвагу
да
старасветчыны
,
да

кнігі
на
Берасцейшчыне
.
З
Санкт-Пецярбурга
прыедзе
доктар

навук
Мікалай
Нікалаеў
,
загадчык

.

—

.
З

—

.

імя

.

.

.
