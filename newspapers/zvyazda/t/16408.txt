.

,

.
Месца

,

,

,

.

?
Анатоль
,

.
Орша

звярнуцца
з

.
Гэта
ўстаноўлена

1.6
пастановы
Савета
Міністраў
Рэспублікі
Беларусь
ад
10.12.1997
г
.
№
1635
«
Аб
працоўных
кніжках
работнікаў
»
са
змяненнямі
і
дапаўненнямі
па
стане
на
25.03.2008
г
.
Згодна
з
гэтым
пунктам
,
у
выпадку
страты
працоўнай
кніжкі
(
укладыша
да

)
работнік

па
апошнім

на

года
з
дня

.
Не

за
15

з

выдае
работніку
другую
працоўную
кніжку
(
укладыш
да
яе
)
устаноўленага

1995

з

«
Дублікат
»
.
Пры

гэтага
тэрміну
наймальнікам
выдаецца

аб

працы
ў
яго
работніка
,
якая
з'яўляецца

па
новым

працы
такому
работніку
другой
працоўнай
кніжкі
(
укладыша
да
яе
)
,
але
з
надпісам
«
Дублікат
»
.
Пры

па

аб
перыядзе
працы
ў
гэтага
наймальніка
па
месцы
захоўвання
здадзеных
ва
ўстаноўленым
парадку
дакументаў
строгай
справаздачнасці
(
уласнік
ліквідуемага
наймальніка
,
правапераемнік
,
мясцовы
выканаўчы
камітэт
,
архіў
і
інш
.
)
.
У
Інструкцыі
аб
парадку
вядзення
працоўных
кніжак
работнікаў
,
якая
зацверджана
пастановай
Міністэрства
працы
Рэспублікі
Беларусь
ад
09.03.1998
г
.
№
30
са
змяненнямі
і
дапаўненнямі
,
растлумачваецца
,

.
Згодна
з

53—54
Інструкцыі
пры
запаўненні

ў
раздзелах
«
Звесткі
аб
рабоце
»
,
«
Звесткі
аб
узнагароджаннях
»
,
«
Звесткі
аб
заахвочваннях
»
запісы
аб
рабоце
,
узнагароджаннях
і
заахвочваннях
па
месцы
апошняй
працы
ўносяцца
па
агульных

на
падставе
раней
выдадзеных

(

.
Калі

,

«
Звесткі

»

,

.
Агульны

,
г
.
зн
.

,

.
Пасля

,

.
Калі

,

,

ў

.
У

4

,

,

.
Самі

,

,

.
Наймальнік

,

.
Пры

(

да

)

.
