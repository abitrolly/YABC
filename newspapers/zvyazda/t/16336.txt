.

.

.

Германія

.

Кандыцыянер

)

+

Трохдзверка
:
—

—

)

Калі

Найбуйнейшыя

Прычым

Таму

Пры

На

З

Вось

Вялікую

.
Напісанае

.
Акрамя

,

,

.
Вось

Цалкам

.
Але

.
Калі

жа
фарба
,
што

,

«

»
,

аўтамабіль

аўтамабіля
.
Для

на
эстакаду
.
І
паверце
,
гэтая
ўяўная
цяганіна
вартая
выдаткаванага
на
яе
часу
.
Салон
Скончыўшы
агляд
кузава
,
прыступайце
да
даследавання
салона
,
які
таксама
вельмі
многа
можа
распавесці
аб
біяграфіі
аўтамабіля
.
Аб
інтэнсіўнай
эксплуатацыі
,
а
значыць
немалым
прабегу
(
часта
ў
разы
большым
,
чым
горда
дэманструе
прадавец
на
адометры
)
кажуць
пацёртыя
чахлы
сядзенняў
,
дываны
,
выцертыя
ў
некаторых
месцах
да
асновы
,
недагледжаная
панэль
прыбораў
,
вышмальцаванае
рулявое
кола
,
ручка
пераключэння
перадач
.
Калі
гэта
так
,
то
аўто
праехала
зусім
не
130—170
тысяч
кіламетраў
,
як
вас
будуць
запэўніваць
,
а
250—300
тысяч
.
Да
таго
ж
даўно
вядома
,
што
сярэдні
прабег
прыватнага
аўто
ў
Еўропе
—
20
тысяч
кіламетраў
у
год
.
Вось
і
памножце
на
яго
ўзрост
.
А
калі
машына
належала
фірме
ці
выкарыстоўвалася
пад
таксі
,
такі
прабег
магла
мець
ледзь
не
штомесяц
.
Далей
варта
праверыць
спраўнасць
усіх
клавішаў
,
рычагоў
і
пераключальнікаў
,
размешчаных
у
салоне(асаблівая
ўвага
карэктару
фар
,
абмывальнікам
шкла
)
,
бо
зараз
нават
дробны
рамонт
электраабсталявання
каштуе
немалых
грошай
.
Рухавік
Спраўны
матор
заводзіцца
без
праблем
і
ў
спякоту
,
і
ў
холад
.
Праца
рухавіка
павінна
быць
роўнай
,
без
збояў
,
бо
апошняе
паказвае
на
няспраўнасць
сістэмы
запальвання
.
Калі
пры
працы
рухавіка
з
выхлапной
трубы
ідзе
шызы
дым
,
гэта
паказвае
на
праблему
з
маслаздымнымі
калпачкамі
або
на
знос
поршневых
маслаздымных
кольцаў
.
Абавязкова
трэба
пацікавіцца
якасцю
масла
,
залітага
ў
рухавік
.
Для
гэтага
неабходна
шчупам
дастаць
трохі
масла
і
пераканацца
ў
тым
,
што
яно
не
ўтрымлівае
старонніх
прымесяў
і
мае
стандартны
для
гэтай
тэхнічнай
вадкасці
колер
.
Калі
на
шчупе
маюцца
адкладанні
бэжавага
колеру
,
які
нагадвае
маянэз
,
значыць
,
верагодней
за
ўсё
,
выйшла
са
строю
пракладка
галоўкі
блока
цыліндраў
.
Неабходна
агледзець
рухавік
на
прадмет
вонкавай
цечы
масла
,
прычынай
якой
часта
бываюць
сальнікі
каленвала
,
масланасоса
(
замена
абыдзецца
зусім
не
ў
капейкі
,
а
для
некаторых
мадэляў
давядзецца
здымаць
каробку
перадач
)
.
Ну
,
і
,
натуральна
,
для
таго
каб
пра
рухавік
склалася
поўнае
ўражанне
,
абавязкова
варта
праехаць
на
абраным
аўтамабілі
.
КПП
Каробка
перадач
не
павінна
выдаваць
ніякага
груку
і
храбусцення
пры
непасрэдным
пераключэнні
перадач
.
Для
поўнай
праверкі
КПП
неабходна
,
вядома
,
выехаць
на
трасу
,
і
ў
вас
у
сувязі
з
гэтым
могуць
паўстаць
рознагалоссі
з
прадаўцом
.
Маўляў
,
я
выеду
,
а
маё
месца
зоймуць
.
Як-небудзь
дамаўляйцеся
—
яно
таго
варта
.
Тым
больш
што
толькі
на
трасе
вы
зможаце
ўстанавіць
,
ці
не
вядзе
руль
,
што
,
як
правіла
,
з'яўляецца
сведчаннем
парушэння
геаметрыі
кузава
ў
выніку
сур'ёзнага
ДТЗ
.
Калі
гэта
так
,
то
ад
аўто
лепш
адмовіцца
адразу
,
нягледзячы
на
ўсе
іншыя
плюсы
.
Калі
ваш
выбар
выпаў
на
аўтамабіль
з
аўтаматычнай
каробкай
перадач
,
то
варта
ведаць
,
што
ў
спраўным
«
аўтамаце
»
пераключэнне
адбываецца
без
рыўкоў
і
старонняга
шуму
.
Падвеска
Шматлікія
патэнцыйныя
аўтааматары
не
надаюць
вялікага
значэння
стану
падвескі
.
Гэта
з'яўляецца
вялікай
памылкай
,
бо
ад
працаздольнасці
падвескі
непасрэдна
залежыць
бяспека
руху
аўтамабіля
.
Да
таго
ж
падвескі
сучасных
аўто
—
вельмі
дарагі
і
даволі
ненадзейны
вузел
.
Асабліва
на
нашых
дарогах
.
Аб
стане
шаравых
апор
,
падшыпнікаў
ступіцаў
,
шрусаў
можна
атрымаць
поўнае
ўяўленне
толькі
падчас
руху
.
Выразны
грукат
на
ямах
кажа
аб
«
смерці
»
шаравых
.
Падшыпнікі
ступіцаў
,
якія
патрабуюць
замены
,
выдаюць
сябе
выразным
гулам
,
які
з'яўляецца
толькі
на
некаторым
дыяпазоне
хуткасцяў
.
Пахрустванне
пры
паваротах
ці
крананні
з
месца
паказвае
на
неабходнасць
замены
шрусаў
.
Усе
недахопы
і
праблемныя
месцы
канкрэтнага
аўто
—
нагода
для
таго
,
каб
скінуць
цану
.
Але
калі
іх
вельмі
шмат
ці
ёсць
такія
,
што
«
несумяшчальныя
з
жыццём
»
,
усё
ж
лепш
адмовіцца
ад
аўто
,
нягледзячы
нават
на
«
салодкі
»
цэннік
.
