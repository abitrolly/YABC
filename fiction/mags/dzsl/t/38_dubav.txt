.

.

.

Мінус

,

Шарагоўцы

.
–
Мінус

.
–

,

.

.

,

.

,

*
*
*

,

,

,

,

”

.
Яны

.
Толькі

Слухай
,
Кайлюс

Мы

,

,

,
пабег
у

,

?

,

.

ў

.
Аднойчы

,
і

.
Але

.
–

,

.
–

.
–

.
–
Ты

Мы

.

,

.

*
*
*
Вядома

,

Нечаканая

,

,

пра

.
Вядома
ж
,

.

,

,

.
Пад

Уражаны

.

.

.

*
*
*

.
–

Згадаўшы

.
*
*
*
Мінус

,

”
.
Пра

.
Раніцай

,

,

.

.
