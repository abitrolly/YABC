…
Надзея
,

,

Новага
года
,
як
ніколі
.
Не

,

свята
,
як

,

была
зусім
маладой
.
Усё

мінула
,
і

і
лёгкі

,
і
яшчэ
адзін

,
і

ёлкамі
,

,
падарункамі

свой
вір
,
і
яна
,
як
і
ўсе
,
хадзіла
па
магазінах
,
выбірала
нейкія
сувеніры
,
купляла
прысмакі
,

дома

штучнымі

,

і
гэтае
свята
,
ёлку
—

натуральнай
даўно
адмовілася
,
як

выраслі
дзеці
.
Ці
,
хаця
,
адмовілася
яшчэ
раней
—

не
стала

у
дом

з
марозным

свежасці
і
свята

лясную
ёлку
,
устанаўліваць

,
затым
,

,
выносіць

не
было

.

у

у

у

,

.

а

жа
,
былі
!
Сустрэча
з
Віцем
—
хіба
не
цуд
?

не

,

,

.

.

,

,

.
