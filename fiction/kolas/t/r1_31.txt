XXXI
Вербніца
і
Благавешчанне
ў
гэтым
годзе
прыпалі
на
адзін
дзень
.
На

быў
прыехаць
з
Малевіч
а
.
Мадэст
з
дзякам

бо

гэты

звычаю
зносілі

,

.
Праўда
,
у
Цельшыне
было
многа
такіх
палешукоў
,
як
,
напрыклад
,
дзед
Мікіта
,
да

не

чынам
.
I

старыя

,

не
толькі
не
здавалі
іх
,
а
набіраліся
яшчэ
новых
.
Звычайна
а
.
Мадэст
прыязджаў
напярэдадні
пад
вечар
.
Так
і
сёння
прыехаў
ён

рана
,
калі
яшчэ

паспела

за
дубам
,
што
стаяў
збоку
раз'езда
.
Счакаўшы
з
паўгадзіны
,
з
каплічкі
,
што
стаяла
ў

і
цёмных
могілак
,
панёсся

звон
,

.
Тым

,

I

,

А

?

Звоняць
,

?
–
А
мабуць
,
звоняць
.
I
пасля

Ужо
,
мабуць
,

:
штось

Спавядацісямем
?
–
А

,
шчо

.

,

.

,
і
