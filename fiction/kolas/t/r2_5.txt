.
Падышоў

старшыня

,

Савось
Карэнчык
,

,

,
вяртлявы
хлопец
,
і

з
царквы
Св
.
Магдалены
Майсей

.
Яго

«
Месца

»
.
Гэта

выйшла
з
валаснога
праўлення
,

з

М
.
П
.
на

,
дзе
і

.
Дзяк

да

,

.
Ён

,

такім

,
што

Тарас
Іванавіч

,
стукнуўшы

носам

,
наводзіў
на
яго

і
змушаў

.
Садовіч
і
асабліва
яго

і
чакалі
,

далей
.
То

,

з

перакідаўся
з
імі
адным-двума

,

знаходзілася
,
як
бы
гэтыя
людзі
выпадкам
сышліся

на

,
чакаючы

.
–
Што

?

Іванавіч
.
Ён

,

,
–

ж
было
даць
тон

да
справы
.
–
Найдус
ведае
,
дзе
трэба
хадзіць
,
–
чмыхнуў
сядзелец
.
–
Ды

,

Карэнчык
,
–
Тамара
Аляксееўна
і

заклікала

ісці
,
але
я
не
хацеў

станавіцца

дарогі
.
–
Чуеш
,

Стары
,
–
шапнуў
Садовіч
прыяцелю
,
–
давай
,
брат
,
адціснем
Найдуса
!
–
Ну
,
што
ж
?
Папрабуй
,
–
ціха

Лабановіч
.
Садовіч
,
яшчэ
не
бачачы

,
кашлянуў

,
падаў

,
–

,
прыняў

.
Карэнчык

:
–
Яна
дачулася
,
што
прыйшлі
маладыя

,
і

з

,
–

на

.
Садовіч

Лабановіча
,

азваўся

–
Грунт
ёсць
.
–
Прыйдуць
,
–
пераконана
адазваўся
Скаромны
.
–
Давай
,
Тарас
Іванавіч
,
абразікі
,

не
так

.
Сядзельца

,
і

Іванавіч
,

,

.
Дзеля

ў

.
Тарас
Іванавіч

«
молнія
»
,
запаліў
яшчэ
і

,
і
ў
класе
стала

...
Каля

аб'ядналася

.
Туды

старшыня
.
Увесь

.
Аказалася
,
што

адной

,

.
Адным

,

...
Няма
ў

.
Можа
,

.
Тарас
Іванавіч

:
–
I

з

,

ты
ў

,
а
то

,

.
Шырокага

,

ў

.
Сам

–
На
тое
ж
і

у

,

драмаў
.
Аблаяўшы

«
капшуком
»
,
Тарас
Іванавіч

:
–
Ёсель
!
Зараз
жа
дзесь
у

,
а

момант

.
–
На
,

да

.
Ды

–

!
Ёсель

,

.

Праз

Тамара
Аляксееўна

.

Ясна

.
Кожны

Карэнчык

Ён
